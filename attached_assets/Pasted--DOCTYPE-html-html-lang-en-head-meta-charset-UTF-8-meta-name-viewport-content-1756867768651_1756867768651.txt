    <script>
        function navigateTo(section) {
            console.log(`Navigating to ${section}`);
            // Update sidebar active state
            document.querySelectorAll('nav a').forEach(link => {
                link.classList.remove('bg-blue-50');
                link.style.color = 'var(--light-text-color)';
            });
            
            // Set active state for clicked item
            event.target.closest('a').classList.add('bg-blue-50');
            event.target.closest('a').style.color = 'var(--button-color)';
        }

        // Update current time
        function updateTime() {
            const now = new Date();
            const timeString = now.toLocaleTimeString('en-US', {
                hour12: false,
                hour: '2-digit',
                minute: '2-digit'
            });
            document.getElementById('currentTime').textContent = timeString;
        }

        // Initialize and update time every minute
        updateTime();
        setInterval(updateTime, 60000);

        // Simulate real-time updates
        function simulateRealTimeUpdates() {
            // Update transaction volume
            const transactionElement = document.querySelector('.text-2xl.font-bold:nth-of-type(3)');
            if (transactionElement) {
                const currentValue = parseFloat(transactionElement.textContent.replace(/[^0-9.]/g, ''));
                const newValue = (currentValue + Math.random() * 0.5).toFixed(1);
                transactionElement.textContent = `₦${newValue}M`;
            }
        }

        // Update metrics every 30 seconds
        setInterval(simulateRealTimeUpdates, 30000);
    </script>
